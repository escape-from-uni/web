package io.github.maze11.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.NinePatch;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.utils.NinePatchDrawable;
import io.github.maze11.MazeGame;
import io.github.maze11.systems.gameState.Achievement;
import io.github.maze11.systems.gameState.MetaGameState;

public class AchievementScreen extends ScrollTableScreen {
    public AchievementScreen(MazeGame game, String name) {
        super(game, name);
    }

    @Override
    public Table createTableData() {
        Table table = new Table();
        Label.LabelStyle headingStyle = new Label.LabelStyle(bodyFont, Color.GOLD);
        Label.LabelStyle bodyStyle = new Label.LabelStyle(bodyFont, Color.LIGHT_GRAY);
        NinePatchDrawable cellBackground = new NinePatchDrawable(new NinePatch(new Texture(Gdx.files.internal("ui/uirect.png")),12,12,12,12));
        headingStyle.background = cellBackground;
        bodyStyle.background = cellBackground;
        if (MetaGameState.getHighScores().isEmpty()){
            new Label("No Achievements Yet!", bodyStyle);
        }else{
            table.pad(10).add(new Label("Achievement", headingStyle)).grow();
            table.pad(10).add(new Label("Description",headingStyle)).grow().row();
            for (Achievement achievement : MetaGameState.getAchievements()){
                Label name = new Label(getWrappedText(achievement.name(),20), bodyStyle);
                Label desc = new Label(getWrappedText(achievement.description(),50), bodyStyle);
                table.pad(10).add(name).grow();
                table.pad(10).add(desc).grow().row();
            }
        }
        table.setWidth(500);
        return table;
    }

    /**
     * Automatically adds newlines to text to stop it from flowing over the screen.
     * @param unwrapped the unwrapped text
     * @param width how many characters each line should ideally have
     * @return text with \n characters inserted between words when the number of characters would exceed width.
     */
    private static String getWrappedText(String unwrapped,int width){
        String[] words = unwrapped.split("\\s");
        StringBuilder splitText = new StringBuilder();
        int charactersInLine = 0;
        for (int i = 0; i < words.length; i++) {
            if (charactersInLine + words[i].length() > width){
                splitText.append("\n");
                charactersInLine = 0;
            }
            splitText.append(words[i]);
            charactersInLine += words[i].length();
            if (i < words.length - 1 && charactersInLine != 0){
                splitText.append(" ");
            }
        }
        return splitText.toString();
    }
}
