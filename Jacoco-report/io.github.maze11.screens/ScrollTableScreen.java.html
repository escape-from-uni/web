package io.github.maze11.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.NinePatch;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.ScrollPane;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.ClickListener;
import com.badlogic.gdx.scenes.scene2d.utils.NinePatchDrawable;
import io.github.maze11.MazeGame;
import io.github.maze11.ui.FontGenerator;

/**
 * Used for displaying scrolling sub-tables or lists e.g. high scores or achievements
 */
public abstract class ScrollTableScreen extends BaseMenuScreen {
    private String titleName = "";
    public ScrollTableScreen(MazeGame game, String titleName) {
        super(game);
        this.titleName = titleName;
        System.out.println(titleName + " launched");
        buildUI();
    }

    @Override
    protected void buildUI() {
        Table table = new Table();
        table.setFillParent(true);

        Label.LabelStyle titleStyle = new Label.LabelStyle(titleFont, Color.WHITE);
        Label title = new Label(titleName, titleStyle);

        BitmapFont buttonFont = FontGenerator.generateRobotoFont(24, Color.WHITE, skin);
        TextButton.TextButtonStyle buttonStyle = new TextButton.TextButtonStyle(skin.get(TextButton.TextButtonStyle.class));
        buttonStyle.font = buttonFont;

        //TODO: Change to a "Back" button.
        TextButton returnButton = new TextButton("Return", buttonStyle);
        returnButton.addListener(new ClickListener() {
            @Override
            public void clicked(InputEvent event, float x, float y) {
                game.closeCurrentScreen();
                playMenuClick();
            }
        });
        // vertical stack layout for elements in menu screen
        table.add(title).padBottom(20).row(); // e.g. title with 20px space below
        //Add scrolling subtable containing data
        Table innerTable = createTableData();
        ScrollPane scrollPane = new ScrollPane(innerTable);
        ScrollPane.ScrollPaneStyle style = new ScrollPane.ScrollPaneStyle();
        style.vScrollKnob = new NinePatchDrawable(new NinePatch(new Texture(Gdx.files.internal("ui/uirect.png")),12,12,12,12));
        style.background = new NinePatchDrawable(new NinePatch(new Texture(Gdx.files.internal("ui/uirectdark.png")),12,12,12,12));
        scrollPane.setStyle(style);
        scrollPane.setFadeScrollBars(false);
        scrollPane.setScrollbarsVisible(true);
        table.add(scrollPane).height(400).row();
        table.add(returnButton).width(200).height(60).padBottom(10).row();

        stage.addActor(table);
    }

    @Override
    protected float[] getBackgroundColor() {
        return new float[]{0.1f, 0.1f, 0.2f}; // Dark blue
    }
    public abstract Table createTableData();
}
