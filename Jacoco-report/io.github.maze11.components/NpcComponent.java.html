package io.github.maze11.components;

import com.badlogic.ashley.core.Component;
import com.badlogic.gdx.math.Vector2;

/**
 * Represents a goose, which wanders aimlessly, but attacks the player if they get too close
 */
//TODO: NOT USED AT THE MOMENT.
public class NpcComponent implements Component {
    /**
     * The set of states that the goose can take.
     * The behaviour of the goose is different in each state.
     */
    public enum NpcState {
        SEARCH, CHASE, RETREAT, TALK, WANDER
    }

    /**
     * The animations that the goose can play
     */
    public enum NpcAnimState {
        IDLE_UP,
        IDLE_RIGHT,
        IDLE_DOWN,
        IDLE_LEFT,

        WALK_UP,
        WALK_RIGHT,
        WALK_DOWN,
        WALK_LEFT
    }

    /** The state machine state the surveyor is in */
    public NpcState state = NpcState.SEARCH;
    public NpcAnimState animState = NpcAnimState.IDLE_DOWN;

    /** The speed at which the surveyor moves while chasing the player */
    public float chaseSpeed = 7f;
    /** The speed at which the surveyor moves while running from the player */
    public float retreatSpeed = 7f;
    /** The speed at which the surveyor moves while wandering or searching*/
    public float idleSpeed =  2f;

    /** The maximum distance from the home position within which the surveyor chooses a point to wander towards */
    public float wanderRadius = 3f;
    /** The distance to the wander destination below which the surveyor picks a new destination */
    public float wanderAcceptDistance = 0.2f;
    /** The point towards which the surveyor wanders when in the wander state */
    public Vector2 currentWanderWaypoint;

    /** The distance to the target below which it begins chasing */
    public float detectionRadius = 6f;
    /** If it is chasing and the target goes outside this radius, it stops chasing */
    public float forgetRadius = 12f;

    /** The time the surveyor retreats for after biting the player */
    public float retreatTime = 2f;
    /** How long the surveyor has been retreating for */
    public float retreatTimeElapsed;
    /** The position around which the surveyor idles */
    public Vector2 homePosition;
}
