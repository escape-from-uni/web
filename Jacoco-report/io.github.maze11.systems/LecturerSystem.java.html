package io.github.maze11.systems;

import com.badlogic.ashley.core.ComponentMapper;
import com.badlogic.ashley.core.Entity;
import com.badlogic.ashley.core.Family;
import com.badlogic.gdx.math.MathUtils;
import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.utils.Timer;

import io.github.maze11.components.AnimationComponent;
import io.github.maze11.components.ChaserComponent;
import io.github.maze11.components.ChaserComponent.ChaseAnimState;
import io.github.maze11.components.InteractableComponent;
import io.github.maze11.components.PhysicsComponent;
import io.github.maze11.components.TransformComponent;
import io.github.maze11.fixedStep.FixedStepper;
import io.github.maze11.fixedStep.IteratingFixedStepSystem;
import io.github.maze11.messages.*;
import io.github.maze11.systems.gameState.Achievement;
import io.github.maze11.systems.gameState.MetaGameState;

public class LecturerSystem extends IteratingFixedStepSystem {

    ComponentMapper<ChaserComponent> lecturerMapper = ComponentMapper.getFor(ChaserComponent.class);
    ComponentMapper<TransformComponent> transformMapper = ComponentMapper.getFor(TransformComponent.class);
    ComponentMapper<PhysicsComponent> physicsMapper = ComponentMapper.getFor(PhysicsComponent.class);
    ComponentMapper<InteractableComponent> interactableMapper = ComponentMapper.getFor(InteractableComponent.class);

    ComponentMapper<AnimationComponent> animMapper = ComponentMapper.getFor(AnimationComponent.class);

    private Entity target;
    final MessageListener messageListener;
    private static boolean lectured = false;
    private final MessagePublisher messagePublisher;

    public LecturerSystem(FixedStepper fixedStepper, MessagePublisher publisher) {
        super(
                fixedStepper,
                Family.all(
                        ChaserComponent.class,
                        TransformComponent.class,
                        PhysicsComponent.class,
                        InteractableComponent.class,
                        AnimationComponent.class
                ).get()
        );

        this.messageListener = new MessageListener(publisher);
        this.messagePublisher = publisher;
    }

    public void setTarget(Entity target) {
        this.target = target;
    }

    @Override
    protected void fixedStepProcessEntity(Entity entity, float deltaTime) {

        handleMessages();

        ChaserComponent lecturer = lecturerMapper.get(entity);

        var transform = transformMapper.get(entity);
        var targetTransform = transformMapper.get(target);
        var physics = physicsMapper.get(entity);
        var interactable = interactableMapper.get(entity);

        Vector2 displacement = new Vector2(targetTransform.position).sub(transform.position);

        var data = new ProcessData(entity, deltaTime, lecturer, transform, physics, interactable, targetTransform, displacement);

        // AI behaviour
        switch (lecturer.state) {
            case WANDER -> processWander(data);
            case CHASE -> processChase(data);
            case RETREAT -> processRetreat(data);
        }

        @SuppressWarnings("unchecked")
        AnimationComponent<ChaseAnimState> anim = (AnimationComponent<ChaseAnimState>) animMapper.get(entity);

        Vector2 vel = physics.body.getLinearVelocity();
        ChaseAnimState newAnimState = resolveAnimation(vel, lecturer.animState);

        // Reset animation when switching states
        if (anim.currentState != newAnimState) {
            anim.currentState = newAnimState;
            anim.elapsed = 0f;
        } else {
            anim.elapsed += deltaTime;
        }

        var animation = anim.animations.get(anim.currentState);
        if (animation != null) {
            anim.currentFrame = animation.getKeyFrame(anim.elapsed, true);
        }

        // Save for next frame
        lecturer.animState = newAnimState;
    }

    /**
     * Determines which animation the lecturer should play according to its velocity
     * @param vel The current velocity the lecturer is moving at
     * @param last The last animation: used as a default if the lecturer is not moving
     * @return
     */
    ChaseAnimState resolveAnimation(Vector2 vel, ChaseAnimState last) {
        if (vel.len2() < 0.01f) {
            // Idle version of last direction
            return switch (last) {
                case WALK_UP, IDLE_UP -> ChaseAnimState.IDLE_UP;
                case WALK_RIGHT, IDLE_RIGHT -> ChaseAnimState.IDLE_RIGHT;
                case WALK_DOWN, IDLE_DOWN -> ChaseAnimState.IDLE_DOWN;
                case WALK_LEFT, IDLE_LEFT -> ChaseAnimState.IDLE_LEFT;
            };
        }

        // Movement â†’ direction-based walk
        if (Math.abs(vel.x) > Math.abs(vel.y)) {
            return vel.x > 0 ? ChaseAnimState.WALK_RIGHT : ChaseAnimState.WALK_LEFT;
        } else {
            return vel.y > 0 ? ChaseAnimState.WALK_UP : ChaseAnimState.WALK_DOWN;
        }
    }

    /**
     * Consumes any new messages it received and reacts.
     * If the lecturer has met a player, begin the lecture.
//     */
    private void handleMessages() {
        while (messageListener.hasNext()) {
            var message = messageListener.next();

            if (message.type == MessageType.LECTURE) {
                Entity entity = ((LectureMessage) message).getInteractable();
                var lecturerComponent = lecturerMapper.get(entity);
                enterRetreat(lecturerComponent);
                if (!lectured) {
                    lectured = true;
                    lecturerComponent.chaseSpeed = 0;
                    this.lectureToasts(messagePublisher);
                }
            }
        }
    }

    void lectureToasts(MessagePublisher messagePublisher) {
        PlayerSystem.inCutscene = true;
        ToastMessage[] toasts = {
                new ToastMessage("Greetings student!\nWelcome to the lecture on uhhhhhhhhh...", 4f),
                new ToastMessage("How to: play the game...", 4f),
                new ToastMessage("So uh basically you have to do stuff, and, uh you get points.", 4f),
                new ToastMessage("And then you've got to do it, like, fast. right?", 4f),
                new ToastMessage("So don't get delayed and stuff...\nI think that's all there is to it.", 4f),
                new ToastMessage("And...Have...Fun...(Yawn)....Zzzzz...", 4f),
        };

        float startDelay = 0f;
        float gap = 4f;

        for (int i = 0; i < toasts.length; i++) {
            final ToastMessage msg = toasts[i];
            float delay = startDelay + i * gap;

            Timer.schedule(new Timer.Task() {
                @Override
                public void run() {
                    messagePublisher.publish(msg);
                }
            }, delay);
        }
        Timer.schedule(new Timer.Task() {
            @Override
            public void run() {
                PlayerSystem.inCutscene = false;
                MetaGameState.unlockAchievement(new Achievement("Blah Blah Blah", "Get lectured on very interesting subjects."), messageListener.publisher);
            }
        }, 32);

    }

    /**
     * Perform fixed update processing for a lecturer in the wander state
     */
    private void processWander(ProcessData data) {
        if (magnitudeIsWithin(data.displacementFromTarget, data.lecturer.detectionRadius)) {
            enterChase(data.lecturer);
            return;
        }

        var lecturer = data.lecturer;

        if (lecturer.currentWanderWaypoint == null) {
            lecturer.currentWanderWaypoint = new Vector2();
            randomiseWaypoint(lecturer);
        }
        else if (magnitudeIsWithin(
                data.transform.position.cpy().sub(lecturer.currentWanderWaypoint),
                lecturer.wanderAcceptDistance
        )) {
            randomiseWaypoint(lecturer);
        }

        Vector2 direction = new Vector2(lecturer.currentWanderWaypoint)
                .sub(data.transform.position)
                .nor();

        data.physics.body.setLinearVelocity(direction.scl(lecturer.wanderSpeed));
    }

    /**
     * Perform fixed update processing for a lecturer in the chase state
     */
    private void processChase(ProcessData data) {
//        if (lectured){
//            data.physics.body.setLinearVelocity(new Vector2(0,0));
//            return;
//        }
        if (!magnitudeIsWithin(data.displacementFromTarget, data.lecturer.forgetRadius)) {
            enterWander(data.lecturer);
            return;
        }

        Vector2 velocity = new Vector2(data.displacementFromTarget)
                .nor()
                .scl(data.lecturer.chaseSpeed);

        data.physics.body.setLinearVelocity(velocity);
    }

    /**
     * Perform fixed update processing for a lecturer in the retreat state
     */
    private void processRetreat(ProcessData data) {
//        if (lectured){
//            data.physics.body.setLinearVelocity(new Vector2(0,0));
//            return;
//        }

        data.lecturer.retreatTimeElapsed += data.deltaTime;

        if (data.lecturer.retreatTimeElapsed > data.lecturer.retreatTime) {
            data.interactable.interactionEnabled = true;
            enterChase(data.lecturer);
        }

        Vector2 velocity = new Vector2(data.displacementFromTarget)
                .nor()
                .scl(-data.lecturer.retreatSpeed);

        data.physics.body.setLinearVelocity(velocity);
    }

    /**
     * Switches the lecturer provided to the wander state from whichever state it is currently in
     */
    void enterWander(ChaserComponent lecturer) {
        lecturer.state = ChaserComponent.ChaseState.WANDER;
    }

    /**
     * Switches the lecturer provided to the chase state from whichever state it is currently in
     */
    void enterChase(ChaserComponent lecturer) {
        lecturer.state = ChaserComponent.ChaseState.CHASE;
    }

    /**
     * Switches the lecturer provided to the retreat state from whichever state it is currently in
     */
    void enterRetreat(ChaserComponent lecturer) {
        lecturer.retreatTimeElapsed = 0f;
        lecturer.state = ChaserComponent.ChaseState.RETREAT;
    }

    private boolean magnitudeIsWithin(Vector2 v, float limit) {
        return v.len2() <= limit * limit;
    }

    /**
     * Picks a new waypoint for this lecturer to walk towards
     * @param lecturer The lecturer in need of a waypoint
     */
    void randomiseWaypoint(ChaserComponent lecturer) {
        float dist = (float) Math.sqrt(MathUtils.random()) * lecturer.wanderRadius;

        lecturer.currentWanderWaypoint
                .setToRandomDirection()
                .scl(dist)
                .add(lecturer.homePosition);
    }

    private record ProcessData(
            Entity entity,
            float deltaTime,
            ChaserComponent lecturer,
            TransformComponent transform,
            PhysicsComponent physics,
            InteractableComponent interactable,
            TransformComponent target,
            Vector2 displacementFromTarget
    ) {}
}
