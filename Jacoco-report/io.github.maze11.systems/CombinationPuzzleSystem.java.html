package io.github.maze11.systems;

import com.badlogic.ashley.core.Entity;
import com.badlogic.ashley.core.Family;
import io.github.maze11.components.ComboPuzzleComponent;
import io.github.maze11.components.HiddenWallComponent;
import io.github.maze11.components.InteractableComponent;
import io.github.maze11.components.PlayerComponent;
import io.github.maze11.fixedStep.FixedStepper;
import io.github.maze11.fixedStep.IteratingFixedStepSystem;
import io.github.maze11.messages.*;
import io.github.maze11.systems.gameState.Achievement;
import io.github.maze11.systems.gameState.MetaGameState;

import javax.swing.*;

/**
 * Listens for pressure plates that are pressed, removes hidden walls when a corresponding pressure plate is pressed
 */
public class CombinationPuzzleSystem extends IteratingFixedStepSystem {


    final MessageListener messageListener;

    public CombinationPuzzleSystem(FixedStepper stepper, MessagePublisher publisher) {
        super(stepper, Family.all(ComboPuzzleComponent.class).get());
        this.messageListener = new MessageListener(publisher);
    }

    @Override
    public void fixedUpdate(float deltaTime){
        while (messageListener.hasNext()) {
            var message = messageListener.next();
            System.out.println(message.type);
            // Only interested in open password messages
            if (message.type == MessageType.OPEN_PASSWORD) {
                Entity entity = ((OpenPasswordMessage)message).getInteractable();

                System.out.println("Start Combination Puzzle");
                new Thread(() -> {
                    String answer = JOptionPane.showInputDialog(null, "What is the combination?", null);
                    //puzzle solution: code is the number of items in the room:
                    //6 chairs, 8 Plants, 3 Monitors, 5 Lamps
                    //give points
                    if (answer.equals("6835")){
                        messageListener.publisher.publish(new Message(MessageType.UNLOCK_SAFE));
                        messageListener.publisher.publish(new ToastMessage("Well Done!",4f));
                        MetaGameState.unlockAchievement(new Achievement("Clever","Unlock the safe"),messageListener.publisher);
                    }else{
                        messageListener.publisher.publish(new ToastMessage("That's not quite right...",3f));
                        //re-enable interaction so that you can try again
                        entity.getComponent(InteractableComponent.class).interactionEnabled = true;
                    }

                }).start();
            }
        }
    }

    @Override
    protected void fixedStepProcessEntity(Entity entity, float deltaTime) {
        // No actions need to be taken
    }
}
