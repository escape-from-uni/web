package io.github.maze11.systems;

import com.badlogic.ashley.core.ComponentMapper;
import com.badlogic.ashley.core.Entity;
import com.badlogic.ashley.core.Family;
import com.badlogic.gdx.math.Vector2;
import io.github.maze11.components.*;
import io.github.maze11.fixedStep.FixedStepper;
import io.github.maze11.fixedStep.IteratingFixedStepSystem;
import io.github.maze11.messages.*;

/**
 * Listens for pressure plates that are pressed, removes hidden walls when a corresponding pressure plate is pressed
 */
public class SlipTriggerSystem extends IteratingFixedStepSystem {

    private final ComponentMapper<SlipTriggerComponent> slipMapper = ComponentMapper.getFor(SlipTriggerComponent.class);

    final MessageListener messageListener;
    private static boolean slippedPreviously = false;

    public SlipTriggerSystem(FixedStepper stepper, MessagePublisher publisher) {
        super(stepper, Family.all(SlipTriggerComponent.class).get());
        this.messageListener = new MessageListener(publisher);
    }

    @Override
    public void fixedUpdate(float deltaTime){
        while (messageListener.hasNext()) {
            var message = messageListener.next();
            // Only interested in pressure plate messages
            if (message.type == MessageType.SLIPPERY) {
                SlipMessage slipMessage = (SlipMessage) message;
                if (!slippedPreviously && ((SlipMessage) message).slipEnabled){
                    messageListener.publisher.publish(new ToastMessage("Uh-oh! Your shoes are coated in slippery soapy water, watch where you're going...", 5f));
                    slippedPreviously = true;
                }
                // Looks through the slip triggers, disable any redundant triggers temporarily, allowing player to walk through
                for (Entity entity : getEntities()) {
                    SlipTriggerComponent slip = slipMapper.get(entity);
                    if (slip != null){
                        PhysicsComponent physics = entity.getComponent(PhysicsComponent.class);
                        physics.body.setActive(slipMessage.slipEnabled != slip.triggersSlip);
                        //re-enable interaction to allow for the trigger to be touched again
                        entity.getComponent(InteractableComponent.class).interactionEnabled = true;
                    }
                }
            }
        }
    }

    @Override
    protected void fixedStepProcessEntity(Entity entity, float deltaTime) {
        // Make it so that the slipper always triggers and can't be disabled on first touch
        entity.getComponent(InteractableComponent.class).interactionEnabled = true;
    }
}
