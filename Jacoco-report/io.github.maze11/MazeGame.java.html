package io.github.maze11;

import com.badlogic.gdx.Game;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import io.github.maze11.assetLoading.AssetLoader;
import io.github.maze11.screens.MenuScreen;
import io.github.maze11.systems.gameState.Achievement;
import io.github.maze11.systems.gameState.MetaGameState;
import io.github.maze11.systems.gameState.ScoreCard;

import java.util.ArrayList;
import java.util.List;
import java.util.Stack;

/**
 * Main game class that manages screen switching and shared resources.
 * Extends LibGDX Game class which provides screen management via setScreen().
 *
 * Each Screen is independent and owns its own resources (fonts, UI, ECS)
 */
public class MazeGame extends Game {
    private SpriteBatch batch;
    private AssetLoader assetLoader;
    public static final int PIXELS_TO_UNIT = 32;

    /** Returns the spriteBatch to be used for rendering the game */
    public SpriteBatch getBatch() {
        return batch;
    }

    /** Returns a reference to the AssetLoader which can be used to access all assets */
    public AssetLoader getAssetLoader() { return assetLoader; }

    /** The screens queued for disposal */
    private List<Screen> discontinuedScreens = new ArrayList<Screen>();
    private Stack<Screen> openScreens = new Stack<>();

    @Override
    public void create() {
        System.out.println("Maze game launched");
        batch = new SpriteBatch();

        assetLoader = new AssetLoader();
        assetLoader.load();

        MetaGameState.addHighScore(new ScoreCard(999999999,null,"bob"));
        //MetaGameState.unlockAchievement(new Achievement("very long achievement name wow cool haha","test"));
        //MetaGameState.unlockAchievement(new Achievement("long description test","test test test test test test ahohifeiof aefieaiofhia aiohf test test test test test test test test test test test test test test test test test test test test"));
        openScreen(new MenuScreen(this));
    }


    @Override
    public void render() {
        super.render();
        for (Screen screen : discontinuedScreens) {
            screen.dispose();
        }
        discontinuedScreens.clear();
    }

    @Override
    public void dispose() {
        batch.dispose();
        assetLoader.dispose();
    }

    /**
     * Switches out the screen, and the ENTIRE stack of open windows to the new one
     * and enqueues the old screens for disposal
     * @param newScreen The screen that will replace all screens
     */
    public void replaceScreen(Screen newScreen){
        //remove and dispose all old screens
        while (!openScreens.isEmpty()){
            discontinuedScreens.add(openScreens.pop());
        }
        openScreen(newScreen);
    }
    /**
     * Opens a screen 'on top' of already existing screens. When it is closed by calling closeCurrentScreen(), the
     * screen below it is rendered again. Useful for pause screens, menus, etc. where you might need to go back
     * to the previous screen, or you want to have a huge amount of sub-windows for some reason.
     * @param newScreen The new screen to be added on top of a stack which represents the open screens
     */
    public void openScreen(Screen newScreen){
        openScreens.add(newScreen);
        this.setScreen(newScreen);
        System.out.println("Render "+newScreen.toString());
    }
    /**
     * Closes the screen that is currently rendering on the stack of open screens. The screen below it will become
     * the active screen. If there are no more screens to close, then close the game.
     */
    public void closeCurrentScreen(){
        if (openScreens.size() <= 1){
            System.out.println("No more screens to close, exiting application.");
            System.exit(1);
        }else{
            Screen topScreen = openScreens.pop();
            discontinuedScreens.add(topScreen);
            this.setScreen(openScreens.peek());
            System.out.println("Closed "+topScreen.toString());
        }
    }
}
