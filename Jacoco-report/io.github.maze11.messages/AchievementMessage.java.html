package io.github.maze11.messages;

import io.github.maze11.systems.gameState.Achievement;

/** Requests an achievement to be saved to the game */
public class AchievementMessage extends Message {

    /** The text to display on the toast */
    public final String text;
    /** The duration to display the toast for */
    public final float duration;

    public AchievementMessage(String text, float duration) {
        super(MessageType.ACHIEVEMENT);
        this.text = text;
        this.duration = duration;
    }
}
