package io.github.maze11.systems.gameState;

import io.github.maze11.messages.AchievementMessage;
import io.github.maze11.messages.MessagePublisher;
import io.github.maze11.messages.ToastMessage;

import java.util.Comparator;
import java.util.HashSet;
import java.util.TreeSet;

/**
 * Tracks the out-of-game systems e.g. the high scores
 */
public class MetaGameState {
    private static final int MAX_HIGH_SCORES = 5;

    private static final TreeSet<ScoreCard> highScores = new TreeSet<>(Comparator.comparingInt(ScoreCard::totalScore));
    private static final HashSet<Achievement> achievements = new HashSet<>();
    public static void addHighScore(ScoreCard score) {
        if (highScores.size() < MAX_HIGH_SCORES){
            highScores.add(score);
        }else{
            ScoreCard lowest = highScores.first();
            if (lowest.totalScore() < score.totalScore()){
                highScores.remove(lowest);
                highScores.add(score);
            }
        }
    }
    public static TreeSet<ScoreCard> getHighScores(){
        return highScores;
    }
    public static void unlockAchievement(Achievement achievement, MessagePublisher messagePublisher){
        if (!achievements.contains(achievement)){
            messagePublisher.publish(new AchievementMessage("Unlocked achievement: "+achievement.name(),3));
            achievements.add(achievement);
        }
    }
    public static HashSet<Achievement> getAchievements(){
        return achievements;
    }
}
